using System;
using System.Drawing;
using System.Drawing.Imaging;
using System.IO;
using System.Linq;
using System.Security.Cryptography;
using System.Text;

class Program
{
    static void Main()
    {
        Console.WriteLine("🔒 Steganography Tool v1.0.2 - FIXED");
        Console.WriteLine("1 - Спрятать текст в PNG");
        Console.WriteLine("2 - Извлечь и расшифровать");
        Console.Write("Выбор: ");
        string choice = Console.ReadLine()?.Trim();
        
        if (choice == "1")
        {
            Console.Write("input.png: ");
            string input = Console.ReadLine()?.Trim();
            Console.Write("output.png: ");
            string output = Console.ReadLine()?.Trim();
            Console.Write("Текст: ");
            string text = Console.ReadLine();
            Console.Write("Пароль: ");
            string password = Console.ReadLine();
            
            if (string.IsNullOrEmpty(input) || string.IsNullOrEmpty(output))
            {
                Console.WriteLine("❌ Ошибка: пути обязательны");
                return;
            }
            
            try
            {
                string encrypted = Encrypt(text ?? "", password ?? "");
                var data = Encoding.UTF8.GetBytes(encrypted);
                var lenBytes = BitConverter.GetBytes(data.Length);
                var allBytes = lenBytes.Concat(data).ToArray();
                
                Bitmap bmpIn = new Bitmap(input);
                Bitmap bmp = new Bitmap(bmpIn);
                bmpIn.Dispose();
                Bitmap result = EmbedBytes(allBytes, bmp);
                result.Save(output, ImageFormat.Png);
                bmp.Dispose();
                result.Dispose();
                Console.WriteLine($"✅ Текст зашифрован и спрятан! Длина: {encrypted.Length} символов");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"❌ Ошибка: {ex.Message}");
            }
        }
        else if (choice == "2")
        {
            Console.Write("image.png: ");
            string path = Console.ReadLine()?.Trim();
            Console.Write("Пароль: ");
            string password = Console.ReadLine();
            
            try
            {
                Bitmap bmp = new Bitmap(path ?? "");
                byte[] allBytes = ExtractBytes(bmp);
                bmp.Dispose();
                
                if (allBytes.Length < 4)
                    throw new Exception("Недостаточно данных в изображении");
                
                int len = BitConverter.ToInt32(allBytes, 0);
                if (allBytes.Length < 4 + len)
                    throw new Exception("Повреждённые данные в изображении");
                
                var data = new byte[len];
                Buffer.BlockCopy(allBytes, 4, data, 0, len);
                string encrypted = Encoding.UTF8.GetString(data);
                
                string decrypted = Decrypt(encrypted, password ?? "");
                Console.WriteLine($"✅ {decrypted}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"❌ Ошибка: {ex.Message}");
            }
        }
        else
        {
            Console.WriteLine("❌ Неверный выбор");
        }
        
        Console.WriteLine("\nНажмите Enter...");
        Console.ReadLine();
    }
    
    static string Encrypt(string plainText, string passPhrase)
    {
        byte[] salt = GenerateRandomBytes(32);
        byte[] iv = GenerateRandomBytes(16);
        byte[] plain = Encoding.UTF8.GetBytes(plainText);
        byte[] key;
        
        using (Rfc2898DeriveBytes pdb = new Rfc2898DeriveBytes(passPhrase, salt, 1000))
            key = pdb.GetBytes(32);
        
        using (Aes aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;
            aes.Mode = CipherMode.CBC;
            aes.Padding = PaddingMode.PKCS7;
            
            using (var encryptor = aes.CreateEncryptor())
            using (var ms = new MemoryStream())
            {
                using (var cs = new CryptoStream(ms, encryptor, CryptoStreamMode.Write))
                    cs.Write(plain, 0, plain.Length);
                return Convert.ToBase64String(salt.Concat(iv).Concat(ms.ToArray()).ToArray());
            }
        }
    }
    
    static string Decrypt(string cipherText, string passPhrase)
    {
        byte[] all = Convert.FromBase64String(cipherText);
        byte[] salt = all.Take(32).ToArray();
        byte[] iv = all.Skip(32).Take(16).ToArray();
        byte[] cipher = all.Skip(48).ToArray();
        byte[] key;
        
        using (Rfc2898DeriveBytes pdb = new Rfc2898DeriveBytes(passPhrase, salt, 1000))
            key = pdb.GetBytes(32);
        
        using (Aes aes = Aes.Create())
        {
            aes.Key = key;
            aes.IV = iv;
            aes.Mode = CipherMode.CBC;
            aes.Padding = PaddingMode.PKCS7;
            
            using (var decryptor = aes.CreateDecryptor())
            using (var ms = new MemoryStream(cipher))
            using (var cs = new CryptoStream(ms, decryptor, CryptoStreamMode.Read))
            using (var sr = new StreamReader(cs, Encoding.UTF8))
                return sr.ReadToEnd();
        }
    }
    
    static byte[] GenerateRandomBytes(int length)
    {
        byte[] random = new byte[length];
        using (RNGCryptoServiceProvider rng = new RNGCryptoServiceProvider())
            rng.GetBytes(random);
        return random;
    }
    
    static Bitmap EmbedBytes(byte[] data, Bitmap bmp)
    {
        int dataIndex = 0;
        int bitIndex = 0;
        
        for (int y = 0; y < bmp.Height; y++)
        {
            for (int x = 0; x < bmp.Width; x++)
            {
                if (dataIndex >= data.Length) return bmp;
                
                Color p = bmp.GetPixel(x, y);
                int r = p.R & ~1;
                int g = p.G & ~1;
                int b = p.B & ~1;
                
                for (int n = 0; n < 3 && dataIndex < data.Length; n++)
                {
                    int bit = (data[dataIndex] >> (7 - bitIndex)) & 1;
                    switch (n)
                    {
                        case 0: r |= bit; break;
                        case 1: g |= bit; break;
                        case 2: b |= bit; break;
                    }
                    
                    bitIndex++;
                    if (bitIndex == 8)
                    {
                        bitIndex = 0;
                        dataIndex++;
                    }
                }
                
                bmp.SetPixel(x, y, Color.FromArgb(r, g, b));
            }
        }
        return bmp;
    }
    
    static byte[] ExtractBytes(Bitmap bmp)
    {
        using (var ms = new MemoryStream())
        {
            int currentByte = 0;
            int bitIndex = 0;
            
            for (int y = 0; y < bmp.Height; y++)
            {
                for (int x = 0; x < bmp.Width; x++)
                {
                    Color p = bmp.GetPixel(x, y);
                    
                    int[] bits = { p.R & 1, p.G & 1, p.B & 1 };
                    foreach (var bit in bits)
                    {
                        currentByte = (currentByte << 1) | bit;
                        bitIndex++;
                        
                        if (bitIndex == 8)
                        {
                            ms.WriteByte((byte)currentByte);
                            currentByte = 0;
                            bitIndex = 0;
                        }
                    }
                }
            }
            return ms.ToArray();
        }
    }
}
