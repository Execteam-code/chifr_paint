using System;
using System.Drawing;
using System.Drawing.Imaging;
using System.IO;
using System.Text;

class Program
{
    enum State { HIDING, FILL_WITH_ZEROS };
    
    static void Main()
    {
        Console.WriteLine("Steganography Tool v0.5");
        Console.WriteLine("1 - Шифровка");
        Console.WriteLine("2 - Расшифровка");
        string choice = Console.ReadLine();
        
        if (choice == "1")
        {
            Console.Write("input.png: ");
            string input = Console.ReadLine();
            Console.Write("output.png: ");
            string output = Console.ReadLine();
            Console.Write("Строка: ");
            string text = Console.ReadLine();
            Console.Write("Пароль: ");
            string password = Console.ReadLine();
            
            string encrypted = EncryptXOR(text, password);
            Bitmap bmp = new Bitmap(input);
            Bitmap result = EmbedText(encrypted, bmp);
            result.Save(output, ImageFormat.Png);
            Console.WriteLine("Зашифровано в output.png");
        }
        else if (choice == "2")
        {
            Console.Write("image.png: ");
            string path = Console.ReadLine();
            Console.Write("Пароль: ");
            string password = Console.ReadLine();
            
            Bitmap bmp = new Bitmap(path);
            string encrypted = ExtractText(bmp);
            string decrypted = DecryptXOR(encrypted, password);
            Console.WriteLine($"Расшифровано: {decrypted}");
        }
    }
    
    static string EncryptXOR(string text, string key)
    {
        byte[] data = Encoding.UTF8.GetBytes(text);
        byte[] keyBytes = Encoding.UTF8.GetBytes(key);
        byte[] result = new byte[data.Length];
        for (int i = 0; i < data.Length; i++)
            result[i] = (byte)(data[i] ^ keyBytes[i % keyBytes.Length]);
        return Convert.ToBase64String(result);
    }
    
    static string DecryptXOR(string encrypted, string key)
    {
        byte[] data = Convert.FromBase64String(encrypted);
        byte[] keyBytes = Encoding.UTF8.GetBytes(key);
        byte[] result = new byte[data.Length];
        for (int i = 0; i < data.Length; i++)
            result[i] = (byte)(data[i] ^ keyBytes[i % keyBytes.Length]);
        return Encoding.UTF8.GetString(result);
    }
    
    static Bitmap EmbedText(string text, Bitmap bmp)
    {
        State state = State.HIDING;
        int charIndex = 0;
        int charValue = 0;
        long colorUnitIndex = 0;
        int zeros = 0;
        
        for (int i = 0; i < bmp.Height; i++)
        {
            for (int j = 0; j < bmp.Width; j++)
            {
                Color pixel = bmp.GetPixel(j, i);
                int R = pixel.R - pixel.R % 2;
                int G = pixel.G - pixel.G % 2;
                int B = pixel.B - pixel.B % 2;
                
                for (int n = 0; n < 3; n++)
                {
                    if (colorUnitIndex % 8 == 0)
                    {
                        if (zeros == 8) return bmp;
                        if (charIndex >= text.Length)
                            state = State.FILL_WITH_ZEROS;
                        else
                            charValue = text[charIndex++];
                    }
                    
                    switch (colorUnitIndex % 3)
                    {
                        case 0: R += charValue % 2; charValue /= 2; break;
                        case 1: G += charValue % 2; charValue /= 2; break;
                        case 2: B += charValue % 2; charValue /= 2; bmp.SetPixel(j, i, Color.FromArgb(R, G, B)); break;
                    }
                    colorUnitIndex++;
                    if (state == State.FILL_WITH_ZEROS) zeros++;
                }
            }
        }
        return bmp;
    }
    
    static string ExtractText(Bitmap bmp)
    {
        int colorUnitIndex = 0;
        int charValue = 0;
        string result = "";
        
        for (int i = 0; i < bmp.Height; i++)
        {
            for (int j = 0; j < bmp.Width; j++)
            {
                Color pixel = bmp.GetPixel(j, i);
                for (int n = 0; n < 3; n++)
                {
                    switch (colorUnitIndex % 3)
                    {
                        case 0: charValue = charValue * 2 + pixel.R % 2; break;
                        case 1: charValue = charValue * 2 + pixel.G % 2; break;
                        case 2: charValue = charValue * 2 + pixel.B % 2; break;
                    }
                    colorUnitIndex++;
                    if (colorUnitIndex % 8 == 0)
                    {
                        charValue = ReverseBits(charValue);
                        if (charValue == 0) break;
                        result += (char)charValue;
                    }
                }
            }
        }
        return result;
    }
    
    static int ReverseBits(int n)
    {
        int result = 0;
        for (int i = 0; i < 8; i++)
        {
            result = result * 2 + n % 2;
            n /= 2;
        }
        return result;
    }
}
